#include <iostream>
#include <string>
using namespace std;

struct MenuItem {
    string name;
    int price;
};

void bubbleSortAsc(MenuItem arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j].price > arr[j + 1].price) {
                MenuItem temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

void bubbleSortDesc(MenuItem arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j].price < arr[j + 1].price) {
                MenuItem temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

void printMenu(MenuItem arr[], int n) {
    for (int i = 0; i < n; i++) {
        cout << arr[i].name << " - Rp" << arr[i].price << endl;
    }
}

int main() {
    const int numFood = 10;
    const int numDrink = 10;

    MenuItem foods[numFood] = {
        {"mie tibal", 10000},
        {"rambut nenek", 6000},
        {"brownies loomer", 8000},
        {"onion ring", 10000},
        {"pao-pao", 10000},
        {"moza stick", 12000},
        {"brokoli krispi", 12000},
        {"ajam katsu setitik", 15000},
        {"tastie croffle", 15000},
        {"klapertaart", 8000}
    };

    MenuItem drinks[numDrink] = {
        {"footloose", 13000},
        {"yakult leci", 15000},
        {"fresh sour squash", 15000},
        {"es kopsu witte-koekjis", 15000},
        {"afogatto", 18000},
        {"choco berry", 13000},
        {"black tea", 10000},
        {"es merdeka", 18000},
        {"red potion", 18000},
        {"the supernova", 18000}
    };

    int choice;
    cout << "Pilih kategori:\n";
    cout << "1. Makanan\n";
    cout << "2. Minuman\n";
    cout << "Masukkan pilihan: ";
    cin >> choice;

    int option;
    cout << "\nPilih pengurutan:\n";
    cout << "1. Ascending\n";
    cout << "2. Descending\n";
    cout << "Masukkan pilihan: ";
    cin >> option;

    cout << "\nMenu setelah diurutkan berdasarkan harga:\n";
    if (choice == 1) {
        if (option == 1) {
            bubbleSortAsc(foods, numFood);
            printMenu(foods, numFood);
        } else if (option == 2) {
            bubbleSortDesc(foods, numFood);
            printMenu(foods, numFood);
        } else {
            cout << "Pilihan tidak valid.\n";
        }
    } else if (choice == 2) {
        if (option == 1) {
            bubbleSortAsc(drinks, numDrink);
            printMenu(drinks, numDrink);
        } else if (option == 2) {
            bubbleSortDesc(drinks, numDrink);
            printMenu(drinks, numDrink);
        } else {
            cout << "Pilihan tidak valid.\n";
        }
    } else {
        cout << "Pilihan tidak valid.\n";
    }

    return 0;
}
